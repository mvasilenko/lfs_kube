create configmap

```
mkdir primary
echo c > primary/cyan
echo m > primary/magenta
echo y > primary/yellow
echo k > primary/black
echo "known as key" >> primary/black
echo blue > favorite

kubectl create configmap colors --from-literal=text=black --from-file=./favorite --from-file=./primary/
kubectl get configmap colors -o yaml
```

create pod with env vars from configmap

```
apiVersion: v1
kind: Pod
metadata:
  name: shell-env-demo
spec:
  containers:
  - image: nginx
    name: nginx
    env:
      - name: ilike
        valueFrom:
          configMapKeyRef:
            name: colors
            key: favorite
```
